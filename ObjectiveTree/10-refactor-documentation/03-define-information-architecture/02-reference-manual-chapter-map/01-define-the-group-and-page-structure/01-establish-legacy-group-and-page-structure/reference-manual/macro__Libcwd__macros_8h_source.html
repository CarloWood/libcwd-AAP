61// This is debugging libcwd itself.
62#ifndef LIBCWD_LibcwDoutScopeBegin_MARKER
63#if CWDEBUG_DEBUGOUTPUT
64#include <sys/types.h>
65extern "C" ssize_t write(int fd, const void *buf, size_t count);
66#define LIBCWD_STR1(x) #x
67#define LIBCWD_STR2(x) LIBCWD_STR1(x)
68#define LIBCWD_STR3 "LibcwDout at " __FILE__ ":" LIBCWD_STR2(__LINE__) "\n"
69#define LIBCWD_LibcwDoutScopeBegin_MARKER LibcwDebugThreads( ++__libcwd_tsd.internal_debugging_code ); do { size_t __attribute__((unused)) __libcwd_len = ::write(2, LIBCWD_STR3, sizeof(LIBCWD_STR3) - 1); } while(0); LibcwDebugThreads( --__libcwd_tsd.internal_debugging_code )
70#else // !CWDEBUG_DEBUGOUTPUT
71#define LIBCWD_LibcwDoutScopeBegin_MARKER
72#endif // !CWDEBUG_DEBUGOUTPUT
73#endif // !LIBCWD_LibcwDoutScopeBegin_MARKER

75// If this macro is defined, it is prepended immediately before any `os << ...stuff...` code that writes debug output.
76// This can be used to add operator<<'s that are exclusively for debugging (most notably those that print things
77// defined in namespace std).
78#ifndef LIBCWD_USING_OSTREAM_PRELUDE
79#define LIBCWD_USING_OSTREAM_PRELUDE
80#endif
81
90#define LibcwDout( dc_namespace, debug_obj, cntrl, ... ) \
91 LibcwDoutScopeBegin(dc_namespace, debug_obj, cntrl) \
92 LibcwDoutStream << __VA_ARGS__; \
93 LibcwDoutScopeEnd
94
95#define LibcwDoutScopeBegin( dc_namespace, debug_obj, cntrl ) \
96 do \
97 { \
98 LIBCWD_TSD_DECLARATION; \
99 LIBCWD_ASSERT_NOT_INTERNAL; \
100 LIBCWD_LibcwDoutScopeBegin_MARKER; \
101 if (LIBCWD_DO_TSD_MEMBER_OFF(debug_obj) < 0) \
102 { \
103 using namespace ::libcwd; \
104 ::libcwd::channel_set_bootstrap_st __libcwd_channel_set(LIBCWD_DO_TSD(debug_obj) LIBCWD_COMMA_TSD); \
105 bool on; \
106 { \
107 using namespace dc_namespace; \
108 on = (__libcwd_channel_set|cntrl).on; \
109 } \
110 if (on) \
111 { \
112 ::libcwd::debug_ct& __libcwd_debug_object(debug_obj); \
113 LIBCWD_DO_TSD(__libcwd_debug_object).start(__libcwd_debug_object, __libcwd_channel_set LIBCWD_COMMA_TSD); \
114 LIBCWD_USING_OSTREAM_PRELUDE
115
116// Note: LibcwDoutStream is *not* equal to the ostream that was set with set_ostream. It is a temporary stringstream.
117#define LibcwDoutStream \
118 (*LIBCWD_DO_TSD_MEMBER(__libcwd_debug_object, current_bufferstream))
119
120#define LibcwDoutScopeEnd \
121 LIBCWD_DO_TSD(__libcwd_debug_object).finish(__libcwd_debug_object, __libcwd_channel_set LIBCWD_COMMA_TSD); \
122 } \
123 } \
124 } while(0)
