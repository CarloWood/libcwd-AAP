50template<class T>
51 void debug_ct::set_ostream(std::ostream* os, T* mutex)
52 {
53 LIBCWD_TSD_DECLARATION;
54 _private_::set_alloc_checking_off(LIBCWD_TSD);
55 _private_::lock_interface_base_ct* new_mutex = new _private_::lock_interface_tct<T>(mutex);
56 _private_::set_alloc_checking_on(LIBCWD_TSD);
57 LIBCWD_DEFER_CANCEL;
58 _private_::mutex_tct<_private_::set_ostream_instance>::lock();
59 _private_::lock_interface_base_ct* old_mutex = M_mutex;
60 if (old_mutex)
61 old_mutex->lock(); // Make sure all other threads left this critical area.
62 M_mutex = new_mutex;
63 if (old_mutex)
64 old_mutex->unlock();
65 private_set_ostream(os);
66 _private_::mutex_tct<_private_::set_ostream_instance>::unlock();
67 // Delete old_mutex after unlocking in order to avoid a dead lock in case the delete causes debug output.
68 if (old_mutex)
69 {
70 _private_::set_alloc_checking_off(LIBCWD_TSD);
71 delete old_mutex;
72 _private_::set_alloc_checking_on(LIBCWD_TSD);
73 }
74 LIBCWD_RESTORE_CANCEL;
75 }
