28class refcnt_charptr_ct {
29private:
30 int M_reference_count; // Number of smart_ptr objects pointing to this stub.
31 char* M_ptr; // Pointer to the actual character string, allocated with new char[...].
32public:
33 refcnt_charptr_ct(char* ptr) : M_reference_count(1), M_ptr(ptr) { }
34 void increment() { ++M_reference_count; }
35 bool decrement()
36 {
37 if (M_ptr && --M_reference_count == 0)
38 {
39 delete [] M_ptr;
40 M_ptr = NULL;
41 return true;
42 }
43 return false;
44 }
45 char* get() const { return M_ptr; }
46 int reference_count() const { return M_reference_count; }
47};

49class smart_ptr {

54public:
